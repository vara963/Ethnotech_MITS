create table cust (
    cus_id int primary key,
    name varchar(30),
    city varchar(30)
);

create table account(
    acc_id int primary key,
    cus_id int,
    balance decimal(10,2),
    foreign key(cus_id) references cust(cus_id)
);

CREATE TABLE employee (
    e_id INT PRIMARY KEY AUTO_INCREMENT,
    e_name VARCHAR(30),
    salary INT
);

CREATE TABLE emp_msg (
    message VARCHAR(100)
);

DELIMITER //

CREATE TRIGGER employee_insert_trigger
AFTER INSERT
ON employee
FOR EACH ROW
BEGIN
    INSERT INTO emp_msg
    VALUES (CONCAT('New Employee Added: ', NEW.e_name));
END //

DELIMITER ;
